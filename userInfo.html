async function getUserInfo() { try { const response = await fetch("https://full-stack-6-889f.onrender.com/userInfo", { method: "GET", credentials: "include", }); if (!response.ok) { throw new Error("Failed to fetch user info"); } const userInfo = await response.json(); console.log(userInfo); document.getElementById("username").textContent = userInfo.userName; document.getElementById("createdAt").textContent = new Date(userInfo.createdAt).toLocaleString(); } catch (error) { console.error("Error fetching user info:", error); } } document.addEventListener("DOMContentLoaded", getUserInfo);
